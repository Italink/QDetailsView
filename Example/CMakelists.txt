
file(GLOB_RECURSE PROJECT_SOURCE FILES ${CMAKE_CURRENT_SOURCE_DIR}/*.h  ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp )

source_group(TREE ${CMAKE_CURRENT_SOURCE_DIR} FILES ${PROJECT_SOURCE})

add_executable(QDetailsViewExample
    ${PROJECT_SOURCE}
)

set_property(TARGET QDetailsViewExample PROPERTY AUTOMOC ON)

find_package(Qt6 REQUIRED COMPONENTS QuickControls2)

target_link_libraries(QDetailsViewExample PUBLIC QDetailsView Qt::Widgets 	Qt::QuickWidgets 
	Qt::QuickPrivate
	Qt::QuickTemplates2
	Qt::QuickTemplates2Private
    Qt::QuickControls2

)

set_target_properties(
    QDetailsView QDetailsViewExample
    PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/Binaries"  
    LIBRARY_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/Binaries"  
    ARCHIVE_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/Lib"  
)
